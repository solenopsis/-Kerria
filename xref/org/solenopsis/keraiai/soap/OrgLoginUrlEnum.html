17  package org.solenopsis.keraiai.soap;
18  
19  /**
20   * Denotes the login URL for sandboxes and production orgs.
21   *
22   * @author Scot P. Floess
23   */
24  public enum OrgLoginUrlEnum {
25      SANDBOX("https://test.salesforce.com"),
26      PRODUCTION("https://login.salesforce.com");
27  
28      /**
29       * The URL for login.
30       */
31      private final String loginUrl;
32  
33      /**
34       * This constructor sets login URL.
35       *
36       * @param loginUrl the login URL.
37       */
38      private OrgLoginUrlEnum(final String loginUrl) {
39          this.loginUrl = loginUrl;
40      }
41  
42      /**
43       * Return the login URL.
44       *
45       * @return the login URL.
46       */
47      public String getLoginUrl() {
48          return loginUrl;
49      }
50  }
