17  package org.solenopsis.keraiai.soap.credentials;
18  
19  import org.flossware.jcore.AbstractCommonBase;
20  
21  /**
22   * Abstract base class for credentials.
23   *
24   * @author sfloess
25   */
26  public abstract class AbstractCredentials extends AbstractCommonBase implements Credentials {
27  
28      /**
29       * Tests a credentials object for equality.
30       *
31       * @param credentials to compare self to.
32       *
33       * @return true if same reference as self or members are equal.
34       */
35      boolean equalsCredentials(final Credentials credentials) {
36          return null == credentials ? false
37                  : this == credentials ? true
38                          : getUserName().equals(credentials.getUserName())
39                          && getPassword().equals(credentials.getPassword())
40                          && getToken().equals(credentials.getToken())
41                          && getUrl().equals(credentials.getUrl())
42                          && getApiVersion().equals(credentials.getApiVersion());
43      }
44  
45      /**
46       * Default constructor.
47       */
48      protected AbstractCredentials() {
49      }
50  
51      /**
52       * {@inheritDoc}
53       */
54      @Override
55      public boolean equals(final Object object) {
56          return object instanceof Credentials && hashCode() == object.hashCode() ? equalsCredentials((Credentials) object) : false;
57      }
58  
59      /**
60       * {@inheritDoc}
61       */
62      @Override
63      public int hashCode() {
64          return getUserName().hashCode()
65                  + getPassword().hashCode()
66                  + getToken().hashCode()
67                  + getUrl().hashCode()
68                  + getApiVersion().hashCode();
69      }
70  
71      /**
72       * {@inheritDoc}
73       */
74      @Override
75      public final String getSecurityPassword() {
76          return CredentialsUtils.computeSecurityPassword(getPassword(), getToken());
77      }
78  
79      /**
80       * {@inheritDoc}
81       */
82      @Override
83      public String toString() {
84          final StringBuilder sb = new StringBuilder();
85  
86          sb.append("apiVersion       [").append(getApiVersion()).append("]\n");
87          sb.append("password         [").append(getPassword()).append("]\n");
88          sb.append("securityPassword [").append(getSecurityPassword()).append("]\n");
89          sb.append("token            [").append(getToken()).append("]\n");
90          sb.append("url              [").append(getUrl()).append("]\n");
91          sb.append("userName         [").append(getUserName()).append("]");
92  
93          return sb.toString();
94      }
95  }
