17  package org.solenopsis.keraiai.soap.port;
18  
19  import java.util.logging.Level;
20  import javax.xml.ws.Service;
21  import org.flossware.jcore.AbstractCommonBase;
22  import org.flossware.jcore.utils.ObjectUtils;
23  import org.solenopsis.keraiai.soap.WebServiceTypeEnum;
24  
25  /**
26   * Abstract base class for create ports. We will use two type of ports: one that is simply a straight on port and one that is for
27   * sessions.
28   *
29   * Ports consist of a server URL plus a partial URL as denoted in WebServiceTypeEnum and a service name. For API based web services
30   * like the Enterprise or Metadata it is an API version whereas Custom web services use the Apex class name (which can be
31   * extrapolated from the Port getter method on a wsimport generated web service).
32   *
33   * @author Scot P. Floess
34   */
35  public abstract class AbstractPortFactory extends AbstractCommonBase {
36  
37      /**
38       * Default constructor.
39       */
40      protected AbstractPortFactory() {
41      }
42  
43      /**
44       * Create a usable web service port.
45       *
46       * @param <P> the port type desired.
47       *
48       * @param webServiceType the type of web service.
49       * @param serverUrl the SFDC URL to communicate with on web service calls.
50       * @param service the actual Service subclass.
51       * @param portType the port type for the service.
52       * @param name the name of the web service. For API based web services it will represent that number whereas for Custom web
53       * services it is the name of the Apex class.
54       *
55       * @return a usable port to call on a web service.
56       */
57      protected <P> P createPort(WebServiceTypeEnum webServiceType, final String serverUrl, Service service, Class<P> portType, String name) {
58          return logAndReturn(
59                  Level.FINEST, "Created port [{0}] for WebServiceTypeEnum [{1}], serverUrl [{2}], service [{3}], name [{4}]",
60                  UrlUtils.setUrl(ObjectUtils.ensureObject(service, "Must present a service").getPort(portType), serverUrl, webServiceType.getPartialUrl(), name),
61                  webServiceType, serverUrl, service, name
62          );
63      }
64  
65      /**
66       * Create a port that has the session id in the SOAP header and is ready to use.
67       *
68       * @param <P> the port type desired.
69       *
70       * @param webServiceType the type of web service.
71       * @param serverUrl the SFDC URL to communicate with on web service calls.
72       * @param service the actual Service subclass.
73       * @param portType the port type for the service.
74       * @param name the name of the web service. For API based web services it will represent that number whereas for Custom web
75       * services it is the name of the Apex class.
76       * @param sessionId the session id needed to call the Salesforce web service.
77       *
78       * @return a usable port to call on a web service that has <code>sessionId</code> in the SOAP header.
79       */
80      protected <P> P createSessionPort(WebServiceTypeEnum webServiceType, final String serverUrl, final Service service, final Class<P> portType, final String name, final String sessionId) {
81          return logAndReturn(
82                  Level.FINEST, "Created session port [{0}] for WebServiceTypeEnum[{0}], serverUrl [{1}], service [{2}], portType [{3}], name [{4}], sessionId [{5}]",
83                  SessionIdUtils.setSessionId(service, createPort(webServiceType, serverUrl, service, portType, name), sessionId),
84                  webServiceType, serverUrl, service, name, sessionId
85          );
86      }
87  }
