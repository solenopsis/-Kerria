17  package org.solenopsis.keraiai.soap.port.session.proxy;
18  
19  import java.lang.reflect.Proxy;
20  import java.util.logging.Level;
21  import javax.xml.ws.Service;
22  import org.flossware.jcore.utils.ObjectUtils;
23  import org.solenopsis.keraiai.soap.WebServiceTypeEnum;
24  import org.solenopsis.keraiai.soap.port.session.AbstractSessionPortFactory;
25  import org.solenopsis.keraiai.soap.security.LoginContext;
26  import org.solenopsis.keraiai.soap.security.SecurityMgr;
27  
28  /**
29   *
30   * This class is used to make method calls on web service ports and acts as a proxy. It performs auto logins, relogins as well as
31   * retrying SFDC calls when they fail.
32   *
33   * @author Scot P. Floess
34   *
35   */
36  public class ProxiedSessionPortFactory extends AbstractSessionPortFactory {
37  
38      /**
39       * Create a port that has the session id in the SOAP header and is ready to use.
40       *
41       * @param <P> the port type desired.
42       *
43       * @param webServiceType the type of web service.
44       * @param service the actual Service subclass.
45       * @param portType the port type for the service.
46       * @param name the name of the web service. For API based web services it will represent that number whereas for Custom web
47       * services it is the name of the Apex class.
48       * @param loginContext contains the session id needed to call the Salesforce web service.
49       *
50       * @return a usable port to call on a web service that has <code>sessionId</code> in the SOAP header.
51       */
52      <P> P doCreateSessionPort(WebServiceTypeEnum webServiceType, final Service service, final Class<P> portType, final String name, final LoginContext loginContext) {
53          ObjectUtils.ensureObject(webServiceType, "Should have a web service type");
54          ObjectUtils.ensureObject(service, "Should have a service");
55          ObjectUtils.ensureObject(portType, "Should have a port type");
56          ObjectUtils.ensureObject(loginContext, "Should have a login context");
57  
58          return logAndReturn(Level.FINEST, "Created for a proxy port [{0}]", createSessionPort(webServiceType, getSecurityMgr().getSession().getServerUrl(), service, portType, name, loginContext.getSessionId()));
59      }
60  
61      /**
62       * {@inheritDoc}
63       */
64      @Override
65      protected <P> P createSessionPort(WebServiceTypeEnum webServiceType, Service service, Class<P> portType, String name) {
66          return (P) logAndReturn(Level.FINEST, "Created for a proxy [{0}]", Proxy.newProxyInstance(getClass().getClassLoader(), new Class[]{portType}, new PortInvocationHandler(this, webServiceType, service, portType, name)));
67      }
68  
69      /**
70       * This constructor sets the security manager.
71       *
72       * @param securityMgr the security manager.
73       *
74       * @throws IllegalArguentException is securityMgr is null.
75       */
76      public ProxiedSessionPortFactory(SecurityMgr securityMgr) {
77          super(securityMgr);
78      }
79  
80  }
