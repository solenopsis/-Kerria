34  /**
35   * Tests the ProxiedSessionPortFactory.
36   *
37   * @author Scot P. Floess
38   */
39  @RunWith(MockitoJUnitRunner.class)
40  public class ProxiedSessionPortFactoryTest {
41  
42      @Mock
43      SecurityMgr securityMgr;
44  
45      StubService service;
46  
47      @Mock
48      LoginContext loginContext;
49  
50      String url;
51  
52      String name;
53  
54      String sessionId;
55  
56      @Mock(extraInterfaces = {BindingProvider.class})
57      Soap port;
58  
59      @Mock
60      Binding binding;
61  
62      ProxiedSessionPortFactory factory;
63  
64      @Before
65      public void setup() {
66          factory = new ProxiedSessionPortFactory(securityMgr);
67  
68          Mockito.when(securityMgr.getSession()).thenReturn(loginContext);
69  
70          url = TestUtils.generateUniqueStr("http://foo.com");
71          Mockito.when(loginContext.getServerUrl()).thenReturn(url);
72  
73          name = TestUtils.generateUniqueStr();
74  
75          sessionId = TestUtils.generateUniqueStr("sessionId");
76          Mockito.when(loginContext.getSessionId()).thenReturn(sessionId);
77  
78          service = new StubService(port);
79  
80          Mockito.when(((BindingProvider) port).getBinding()).thenReturn(binding);
81      }
82  
83      /**
84       * Test the constructor with a null SecurityMgr.
85       */
86      @Test(expected = IllegalArgumentException.class)
87      public void test_constructor_null() {
88          new ProxiedSessionPortFactory(null);
89      }
90  
91      /**
92       * Test the doCreateSessionPort() method - null WebServiceTypeEnum.
93       */
94      @Test(expected = IllegalArgumentException.class)
95      public void test_doCreateSessionPort_null_WebServiceTypeEnum() {
96          factory.doCreateSessionPort(null, service, Soap.class, name, loginContext);
97      }
98  
99      /**
100      * Test the doCreateSessionPort() method - null Service.
101      */
102     @Test(expected = IllegalArgumentException.class)
103     public void test_doCreateSessionPort_null_Service() {
104         factory.doCreateSessionPort(WebServiceTypeEnum.ENTERPRISE_TYPE, null, Soap.class, name, loginContext);
105     }
106 
107     /**
108      * Test the doCreateSessionPort() method - null Port type.
109      */
110     @Test(expected = IllegalArgumentException.class)

112         factory.doCreateSessionPort(WebServiceTypeEnum.ENTERPRISE_TYPE, service, null, name, loginContext);
113     }
114 
115     /**
116      * Test the doCreateSessionPort() method - null login context.
117      */
118     @Test(expected = IllegalArgumentException.class)
119     public void test_doCreateSessionPort_null_LoginContext() {
120         factory.doCreateSessionPort(WebServiceTypeEnum.ENTERPRISE_TYPE, service, Soap.class, name, null);
121     }
122 
123     /**
124      * Test the doCreateSessionPort() method.
125      */
126     @Test
127     public void test_doCreateSessionPort() {
128         Assert.assertNotNull("Should have a created a Soap port", factory.doCreateSessionPort(WebServiceTypeEnum.ENTERPRISE_TYPE, service, Soap.class, name, loginContext));
129     }

161     }
