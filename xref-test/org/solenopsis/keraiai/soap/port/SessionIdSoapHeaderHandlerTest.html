20  import javax.xml.soap.SOAPElement;
21  import javax.xml.soap.SOAPEnvelope;
22  import javax.xml.soap.SOAPException;
23  import javax.xml.soap.SOAPHeader;
24  import javax.xml.soap.SOAPMessage;
25  import javax.xml.soap.SOAPPart;
26  import javax.xml.ws.handler.soap.SOAPMessageContext;
27  import org.flossware.jcore.utils.TestUtils;
28  import org.junit.Assert;
29  import org.junit.Before;
30  import org.junit.Test;
31  import org.junit.runner.RunWith;
32  import org.mockito.Mock;
33  import org.mockito.Mockito;
34  import org.mockito.runners.MockitoJUnitRunner;
35  
36  /**
37   * Tests the SessionIdSoapHeaderHandler class.
38   *
39   * @author Scot P. Floess
40   */
41  @RunWith(MockitoJUnitRunner.class)
42  public class SessionIdSoapHeaderHandlerTest {
43  
44      @Mock
45      SOAPMessageContext msgContext;
46  
47      @Mock
48      SOAPMessage soapMessage;
49  
50      @Mock
51      SOAPPart soapPart;
52  
53      @Mock
54      SOAPEnvelope soapEnvelope;
55  
56      @Mock
57      SOAPHeader soapHeader;
58  
59      @Mock
60      SOAPElement headerSoapElement;
61  
62      @Mock
63      SOAPElement sessionIdSoapElement;
64  
65      @Mock
66      QName serviceName;
67  
68      String namespaceUri;
69      String sessionId;
70  
71      @Before
72      public void setup() throws SOAPException {
73          namespaceUri = TestUtils.generateUniqueStr("namespace", "uri");
74          sessionId = TestUtils.generateUniqueStr("session", "id");
75  
76          Mockito.when(msgContext.getMessage()).thenReturn(soapMessage);
77          Mockito.when(soapMessage.getSOAPPart()).thenReturn(soapPart);
78          Mockito.when(soapPart.getEnvelope()).thenReturn(soapEnvelope);
79          Mockito.when(soapEnvelope.getHeader()).thenReturn(soapHeader);
80          Mockito.when(soapHeader.addChildElement((QName) Mockito.anyObject())).thenReturn(headerSoapElement);
81          Mockito.when(headerSoapElement.addChildElement(Mockito.anyString())).thenReturn(sessionIdSoapElement);
82  
83          Mockito.when(serviceName.getNamespaceURI()).thenReturn(namespaceUri);
84      }
85  
86      /**
87       * Test a null session header name.
88       */
89      @Test(expected = IllegalArgumentException.class)
90      public void test_constructor_sessionHeaderName_null() {
91          new SessionIdSoapHeaderHandler(null, TestUtils.generateUniqueStr());
92      }
93  
94      /**
95       * Test a null sessionId.
96       */
97      @Test(expected = IllegalArgumentException.class)
98      public void test_constructor_sessionId_null() {
99          new SessionIdSoapHeaderHandler(new QName(TestUtils.generateUniqueStr(), TestUtils.generateUniqueStr()), null);
100     }
101 
102     /**
103      * Test a blank sessionId.
104      */
105     @Test(expected = IllegalArgumentException.class)
106     public void test_constructor_sessionId_blank() {
107         new SessionIdSoapHeaderHandler(new QName(TestUtils.generateUniqueStr(), TestUtils.generateUniqueStr()), "     ");
108     }
109 
110     /**
111      * Test an empty sessionId.
112      */
113     @Test(expected = IllegalArgumentException.class)
114     public void test_constructor_sessionId_empty() {
115         new SessionIdSoapHeaderHandler(new QName(TestUtils.generateUniqueStr(), TestUtils.generateUniqueStr()), "");
116     }
117 
118     /**
119      * Test constructing.
120      */
121     @Test
122     public void test_constructor() {
123         final SessionIdSoapHeaderHandler sessionIdSoapHeaderHandler = new SessionIdSoapHeaderHandler(serviceName, sessionId);
124 
125         Assert.assertSame("Should be the same session header name", serviceName, sessionIdSoapHeaderHandler.getSessionHeaderName());
126         Assert.assertSame("Should be the same session id", sessionId, sessionIdSoapHeaderHandler.getSessionId());
127     }
128 
129     /**
130      * Test no request.
131      */
132     @Test
133     public void test_handleMessage_noRequest() throws SOAPException {
134         final SessionIdSoapHeaderHandler sessionIdSoapHeaderHandler = new SessionIdSoapHeaderHandler(serviceName, sessionId);
135 
136         Mockito.when(msgContext.get(Mockito.anyString())).thenReturn(null);
137 
138         Assert.assertTrue("Should return true", sessionIdSoapHeaderHandler.handleMessage(msgContext));
139 
140         Mockito.verify(msgContext, Mockito.never()).getMessage();
141         Mockito.verify(msgContext.getMessage(), Mockito.never()).saveChanges();
142         Mockito.verify(soapMessage, Mockito.never()).saveChanges();
143     }
144 
145     /**
146      * Test request.
147      */
148     @Test
149     public void test_handleMessage() throws SOAPException {
150         final SessionIdSoapHeaderHandler sessionIdSoapHeaderHandler = new SessionIdSoapHeaderHandler(serviceName, sessionId);
151 
152         Mockito.when(msgContext.get(Mockito.anyString())).thenReturn(true);
153 
154         Assert.assertTrue("Should return true", sessionIdSoapHeaderHandler.handleMessage(msgContext));
155 
156         Mockito.verify(msgContext, Mockito.atLeastOnce()).getMessage();
157         Mockito.verify(soapMessage).saveChanges();
158         Mockito.verify(soapHeader).addChildElement(serviceName);
159         Mockito.verify(headerSoapElement).addChildElement(SessionIdUtils.SESSION_ID);
160         Mockito.verify(sessionIdSoapElement).addTextNode(sessionId);
161     }
162 
163     /**
164      * Test other default methods for code coverage.
165      */
166     @Test
167     public void test_defaultMethods() {
168         final SessionIdSoapHeaderHandler sessionIdSoapHeaderHandler = new SessionIdSoapHeaderHandler(serviceName, sessionId);
169 
170         Assert.assertNull("Should be null", sessionIdSoapHeaderHandler.getHeaders());
171         Assert.assertTrue(sessionIdSoapHeaderHandler.handleFault(msgContext));
172         sessionIdSoapHeaderHandler.close(msgContext);
173     }
174 }
