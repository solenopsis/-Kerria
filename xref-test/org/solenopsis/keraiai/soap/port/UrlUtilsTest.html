17  package org.solenopsis.keraiai.soap.port;
18  
19  import java.net.MalformedURLException;
20  import java.net.URL;
21  import java.util.HashMap;
22  import java.util.Map;
23  import javax.xml.ws.BindingProvider;
24  import org.junit.Assert;
25  import org.junit.Before;
26  import org.junit.Test;
27  import org.junit.runner.RunWith;
28  import org.mockito.Mock;
29  import org.mockito.Mockito;
30  import org.mockito.runners.MockitoJUnitRunner;
31  
32  /**
33   * Tests the UrlUtils utility class.
34   *
35   * @author Scot P. Floess
36   */
37  @RunWith(MockitoJUnitRunner.class)
38  public class UrlUtilsTest {
39  
40      @Mock
41      BindingProvider port;
42  
43      Map<String, Object> requestContext;
44  
45      @Before
46      public void setup() {
47          requestContext = new HashMap<>();
48  
49          Mockito.when(port.getRequestContext()).thenReturn(requestContext);
50      }
51  
52      /**
53       * Test computing a URL string using strings.
54       */
55      @Test
56      public void test_computeUrlString() {
57          Assert.assertEquals("Should be the same string", "https://foo.com/bar/alpha", UrlUtils.computeUrlString("https", "foo.com", "bar", "alpha"));
58      }
59  
60      /**
61       * Test computing a URL string using a url.
62       */
63      @Test
64      public void test_computeUrlString_url() throws MalformedURLException {
65          Assert.assertEquals("Should be the same string", "https://foo.com/bar/alpha", UrlUtils.computeUrlString(new URL("https://foo.com/not/here/now"), "bar", "alpha"));
66      }
67  
68      /**
69       * Test computing a URL string using a null url where the serverUrl has more subpaths.
70       */
71      @Test(expected = IllegalArgumentException.class)
72      public void test_computeUrlString_paths_null() throws MalformedURLException {
73          UrlUtils.computeUrlString((String) null, "bar", "alpha");
74      }
75  
76      /**
77       * Test computing a URL string using a blank url where the serverUrl has more subpaths.
78       */
79      @Test(expected = IllegalArgumentException.class)
80      public void test_computeUrlString_paths_blank() throws MalformedURLException {
81          UrlUtils.computeUrlString("  ", "bar", "alpha");
82      }
83  
84      /**
85       * Test computing a URL string using an empty url where the serverUrl has more subpaths.
86       */
87      @Test(expected = IllegalArgumentException.class)
88      public void test_computeUrlString_paths_empty() throws MalformedURLException {
89          UrlUtils.computeUrlString("", "bar", "alpha");
90      }
91  
92      /**
93       * Test computing a URL string using a maleformed url where the serverUrl has more subpaths.
94       */
95      @Test(expected = IllegalArgumentException.class)
96      public void test_computeUrlString_paths_malformed() throws MalformedURLException {
97          UrlUtils.computeUrlString("h ttp:_fo", "bar", "alpha");
98      }
99  
100     /**
101      * Test computing a URL string using a proper url where the serverUrl has more subpaths.
102      */
103     @Test
104     public void test_computeUrlString_paths() throws MalformedURLException {
105         Assert.assertEquals("Should be the same string", "http://foo.com/bar/alpha", UrlUtils.computeUrlString("http://foo.com/theta/papapapa", "bar", "alpha"));
106     }
107 
108     /**
109      * Test setting a URL using a null url.
110      */
111     @Test(expected = IllegalArgumentException.class)
112     public void test_setUrl_null() throws MalformedURLException {
113         UrlUtils.setUrl(port, null, "bar", "alpha");
114     }
115 
116     /**
117      * Test setting a URL using a blank url.
118      */
119     @Test(expected = IllegalArgumentException.class)
120     public void test_setUrl_blank() throws MalformedURLException {
121         UrlUtils.setUrl(port, "  ", "bar", "alpha");
122     }
123 
124     /**
125      * Test setting a URL using an empty url.
126      */
127     @Test(expected = IllegalArgumentException.class)
128     public void test_setUrl_empty() throws MalformedURLException {
129         UrlUtils.setUrl(port, "", "bar", "alpha");
130     }
131 
132     /**
133      * Test setting a URL using a maleformed url.
134      */
135     @Test(expected = IllegalArgumentException.class)
136     public void test_setUrl_malformed() throws MalformedURLException {
137         UrlUtils.setUrl(port, "h ttp:_fo", "bar", "alpha");
138     }
139 
140     /**
141      * Test setting a URL using a proper url.
142      */
143     @Test
144     public void test_setUrl() throws MalformedURLException {
145         UrlUtils.setUrl(port, "http://foo.com/theta/apapapapa", "bar", "alpha");
146 
147         Assert.assertFalse("Should contain data", requestContext.isEmpty());
148         Assert.assertEquals("Should only be one element", 1, requestContext.size());
149 
150         Assert.assertEquals("Should be correct url", "http://foo.com/bar/alpha", requestContext.get(BindingProvider.ENDPOINT_ADDRESS_PROPERTY));
151     }
152 }
